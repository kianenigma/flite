Keccak derived functions specified in FIPS-202, SP800-185 and KangarooTwelve.

§Example

let input_a = b"hello world";
let input_b = b"!";
let mut output = [0u8; 32];
hasher.update(input_a);
hasher.update(input_b);
hasher.finalize(&mut output);
